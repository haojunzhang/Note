psql -U username -h localhost -p 5432 dbname

\l

列出Table
\dt

列出User
\du

新增User
CREATE USER username WITH PASSWORD 'xxx';

刪除User
DROP USER username;

修改User密碼
ALTER USER scheduler PASSWORD 'new_password';

授權與撤銷
    連線存取
        GRANT|REVOKE CONNECT TO DATABASE database_name TO|FROM username;
    使用DB
        GRANT|REVOKE USAGE ON SCHEMA database_name TO|FROM username;
    所有表
        GRANT|REVOKE ALL ON ALL TABLES IN SCHEMA schema_name TO|FROM scheduler;
    所有SEQUENCES
        GRANT|REVOKE ALL ON ALL SEQUENCES IN SCHEMA schema_name TO|FROM scheduler;
    特定表
        GRANT|REVOKE ALL ON some_table TO|FROM scheduler;
    特定表只讀,建
        GRANT|REVOKE SELECT,INSERT ON some_table TO|FROM scheduler;
